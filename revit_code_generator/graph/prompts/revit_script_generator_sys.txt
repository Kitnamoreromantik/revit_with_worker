        };

        {
            FamilyInstance col = doc.Create.NewFamilyInstance(
                p, columnType, level, Autodesk.Revit.DB.Structure.StructuralType.Column);
        }

        t.Commit();
    }

    return "Created 3 columns of type: " + columnType.Name;
